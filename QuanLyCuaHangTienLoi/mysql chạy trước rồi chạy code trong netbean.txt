-- 1. Tạo Database với chuẩn bộ mã hỗ trợ đa ngôn ngữ đầy đủ (utf8mb4)
CREATE DATABASE IF NOT EXISTS ChatbotDB 
CHARACTER SET utf8mb4 
COLLATE utf8mb4_unicode_ci;

-- 2. Chỉ định sử dụng Database vừa tạo
USE ChatbotDB;

-- 3. Tạo bảng Sessions (Lưu thông tin mỗi phiên trò chuyện)
CREATE TABLE Sessions (
    SessionID INT AUTO_INCREMENT PRIMARY KEY,
    LanguageCode VARCHAR(10) NOT NULL, -- Lưu mã ngôn ngữ (VD: 'vi', 'en', 'ja')
    StartTime TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 4. Tạo bảng ChatHistory (Lưu trữ chi tiết từng tin nhắn)
CREATE TABLE ChatHistory (
    MessageID INT AUTO_INCREMENT PRIMARY KEY,
    SessionID INT,
    Sender VARCHAR(50) NOT NULL, -- Sẽ lưu giá trị là 'User' hoặc 'Bot'
    MessageText TEXT NOT NULL, -- Chứa văn bản gốc người dùng nhập hoặc bot trả lời (Đa ngôn ngữ)
    TranslatedText TEXT, -- Lưu bản dịch tiếng Anh để mô hình AI xử lý (ẩn với người dùng)
    Timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (SessionID) REFERENCES Sessions(SessionID) ON DELETE CASCADE
)
